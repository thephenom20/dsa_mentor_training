#program1
#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,d;
	cin>>n;
	cout<<"Enter the difference:";
	cin>>d;
	vector<int> arr(n);
	for(int i=0;i<n;++i)
		cin>>arr[i];
   
	unordered_map<int,int> freq;
	for(int num:arr){
		freq[num]++;
		
	}
	for(auto x:freq){
	  
		if(freq[x.first-d]>0)
		cout<<"True"<<" " ;
	}
	return 0;  
}
